# This file contains the manual 'docker run' commands to create the required containers.
# It is recommended to use the 'docker-compose.yml' file instead.

# Command to create the PostgreSQL container:
docker run --name vulnverse-postgres -e POSTGRES_USER=admin -e POSTGRES_PASSWORD=adityA10! -e POSTGRES_DB=vulnverse_db -p 5432:5432 -d postgres

# Command to create the OpenVPN container:
# Note: The volume path is specific to your system's setup.
docker run --name openvpn_server -v /d/VULNverse/vulnverse/openvpn-data:/etc/openvpn -d -p 1194:1194/udp --cap-add=NET_ADMIN kylemanna/openvpn

# Command to apply database migrations:
# This should be run after the database container is started.
.\venv\Scripts\python.exe -m alembic -c alembic.ini upgrade head
